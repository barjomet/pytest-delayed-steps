227 def get_pytest_node_current_param_and_fixture_values(request,
228 params_to_ignore=None):
229 """
230 Returns a dictionary containing all fixtures and parameters values available in a given test `request`.
231  
232 As opposed to `get_pytest_node_current_param_values`, this contains fixture VALUES and non-parametrized fixtures,
233 whereas `get_pytest_node_current_param_values` only contains the parameters.
234  
235 :param request:
236 :param params_to_ignore: an iterable of parameter or fixture names to ignore in the returned dictionary
237 :return:
238 """
239 # Default value
240 if params_to_ignore is None:
241 params_to_ignore = []
242  
243 # List the values of all the test function parameters that matter
244 kwargs = {argname: get_fixture_or_param_value(request, argname)
245 for argname in request.funcargnames
246 if argname not in params_to_ignore}
247  
248 return kwargs

251 def get_fixture_or_param_value(request, fixture_or_param_name):
252 """
253 Returns the value associated with parameter or fixture named `fixture_name`, in provided request context.
254 This is just an easy way to use `getfixturevalue` or `getfuncargvalue` according to whichever is available in
255 current pytest version.
256  
257 Note: it seems that this is the same than `request.node.callspec.params[fixture_or_param_name]` but maybe it is
258 less 'internal' as an api ?
259  
260 :param request:
261 :param fixture_or_param_name:
262 :return:
263 """
264 try:
265 # Pytest 4+ or latest 3.x (to avoid the deprecated warning)
266 return request.getfixturevalue(fixture_or_param_name)
267 except AttributeError:
268 # Pytest 3-
269 return request.getfuncargvalue(fixture_or_param_name)
